Audit a local Iceberg warehouse

Generate fixtures, audit a single skewed table, exit non-zero on CRITICAL.

$ pip install "pyiceberg[pyarrow,sql-sqlite]>=0.7.0"
$ python scripts/gen_fixtures.py --root ./warehouse --clean

$ iceberg-sentry audit \
    --catalog localfs --catalog-root ./warehouse \
    --table fixtures.skewed_v2

Gate a PR with SARIF

name: Iceberg health
on: pull_request
permissions:
  contents: read
  security-events: write
jobs:
  audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Install
        run: curl -sSL https://get.icebergsentry.io/install.sh | bash
      - uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: ${{ secrets.ICEBERG_AUDIT_ROLE }}
          aws-region: us-east-1
      - run: |
          iceberg-sentry audit \
            --catalog glue --namespace finance \
            --policy sentry.yaml \
            --format sarif > iceberg.sarif
      - uses: github/codeql-action/upload-sarif@v3
        with: { sarif_file: iceberg.sarif, category: iceberg-health }

Find orphan files, save manifest, delete with your own tool

$ iceberg-sentry orphans \
    --catalog glue --table finance.transactions \
    --grace-period 6h --format json > orphans.json

$ jq -r '.candidates[].uri' orphans.json \
  | aws s3 rm --recursive --quiet
// safety Sentry only emits a recommendation. The actual deletion is a separate step you control — typically with aws s3 rm or your data governance tool. Files written in the 6-hour grace window are never flagged.

Push metrics to Prometheus from CronJob

apiVersion: batch/v1
kind: CronJob
metadata: { name: iceberg-sentry-daily }
spec:
  schedule: "0 2 * * *"
  jobTemplate:
    spec:
      template:
        spec:
          restartPolicy: Never
          containers:
            - name: sentry
              image: ghcr.io/jaybilgaye/iceberg-sentry:latest
              args:
                - audit
                - --catalog=glue
                - --namespace=finance
                - --format=prometheus
                - --push-gateway=http://prometheus-pushgateway:9091
                - --push-job=iceberg-sentry
